1function [flag, x, y, T] = otg_xy_reallyDumb(v1, d1, kj, kT, ks, kd, dT, Tmin, Tmax, dataVeh, safetyS, safetyD, dt, m, kappa, y0, phi, vx0, ax0, w)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%--------------------------------------------------------------------------
  7%
  8%otg_pspdT_reallyDumb returns the points specified by x, y on the
  9%trajectory in the car coordinate system
 10%   INPUT:
 11%
 12%       v1 = velocity in s direction at the end of the trajectory
 13%       d1 = distance from the center line at the end of the trajectory
 14%
 15%       kj = weight for the jerk functional
 16%       kT = weight for the time
 17%       ks = weight of the longitudinal weight function
 18%       kd = weight of the lateral weight function
 19%
 20%       dT = time interval between two possible end Times T
 21%       Tmin = minimal T > 0
 22%       Tmax = maximal T
 23%
 24%       dataVeh = (3xN) data of other vehivles on the road
 25%           dataVeh(:,i) = [s0i; vi; Ii] s.t.
 26%               s0i = initial distance between obstacle car i and own car
 27%               vi  = velocity of obstacle car i anlong the road
 28%               Ii  = -1/1: -1: on right lane, +1: on the left lane
 29%       safetyS = min. safety distance in s direction
 30%       safetyD = min. safety distance in D direction with sign:
 31%           if Ii == -1: d(t) must be bigger  than safetyD
 32%           if Ii ==  1: d(t) must be smaller than safetyD
 33%       dt = sampling interval for the collision detection
 34%
 35%       ds = distance in s direction between two points
 36%       m = number of points wanted
 37%
 38%       kappa = curvature of the center line
 39%       phi = angle between the x-axis of the car and the tangent on the
 40%           center line at the section between center line and y axis
 41%
 42%   OUTPUT:
 43%       flag = 1/0/-1
 44%           1:  everything ok
 45%           0:  solution was not unique. solution with minimal T is choosen
 46%           -1: no solution: the vectors are full of zeros
 47%       x = x in the car coordinate sytsem
 48%       y = y in the car coordinate sytsem
 49%
 50% See also
 51
 52%% init
 53x = zeros(1,m);
 54y = zeros(1,m);
 55
 56%% converte the state to S, D
 57S = [cos(phi)*vx0,-sin(phi)*w*vx0+cos(phi)*ax0, v1]; %really dirty
 58
 59cosphi_d = -sin(phi)*w;
 60cosphi_dd = -w^2*cos(phi);
 61y0_d = -sin(phi)*vx0;
 62y0_dd = -(cos(phi)*w*vx0+sin(phi)*ax0);
 63
 64D = [-cos(phi)*y0,-(cosphi_d*y0+cos(phi)*y0_d),-(cosphi_dd*y0+2*cosphi_d*y0_d+cos(phi)*y0_dd),d1];
 65
 66
 67%% coefficients
 68[flag, ps, pd, T] = otg_pspdT_reallyDumb(S, D, kj, kT, ks, kd, dT, Tmin, Tmax, dataVeh, safetyS, safetyD, dt);
 69
 70if flag == -1
 71    return;
 72end
 73
 74
 75%% calculate tt
 76tt = linspace(0,T,m);
 77
 78%% calc s and d at the times
 79ss = polyval(ps, tt);
 80dd = polyval(pd, tt);
 81
 82if kappa >0
 83    XY = 1/kappa*[sin(kappa*ss);1-cos(kappa*ss)] + [-sin(kappa*ss);cos(kappa*ss)].*repmat(dd,2,1);
 84else
 85    XY = [ss;zeros(1, m)] + repmat([0;1],1,m).*repmat(dd,2,1);
 86end
 87
 88xy_car = zeros(2,m);
 89
 90R = [cos(phi), -sin(phi); sin(phi) cos(phi)];
 91
 92
 93for k = 1:m
 94    xy_car(:,k) =  R*XY(:,k) + [-sin(phi);cos(phi)]*(D(1));
 95end
 96
 97x = xy_car(1,:);
 98y = xy_car(2,:);
 99
100end
