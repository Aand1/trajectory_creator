1function [flag1, flag2, x, y] = otg_xy_reallyDumb(S, D, kj, kT, ks, kd, dT, Tmin, Tmax, dataVeh, safetyS, safetyD, dt, ds, m, kappa, y0, phi)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%--------------------------------------------------------------------------
  7%
  8%otg_pspdT_reallyDumb returns the points specified by x, y on the
  9%trajectory in the car coordinate system
 10%   INPUT:
 11%
 12%       S = [v0, a0, v1] s.t.
 13%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 14%       D = [d0, d0d, d0dd, d1] s.t.
 15%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 16%
 17%       kj = weight for the jerk functional
 18%       kT = weight for the time
 19%       ks = weight of the longitudinal weight function
 20%       kd = weight of the lateral weight function
 21%
 22%       dT = time interval between two possible end Times T
 23%       Tmin = minimal T
 24%       Tmax = maximal T
 25%
 26%       dataVeh = (3xN) data of other vehivles on the road
 27%           dataVeh(:,i) = [s0i; vi; Ii] s.t.
 28%               s0i = initial distance between obstacle car i and own car
 29%               vi  = velocity of obstacle car i anlong the road
 30%               Ii  = -1/1: -1: on left lane, +1: on the right lane
 31%       safetyS = min. safety distance in s direction
 32%       safetyD = min. safety distance in D direction with sign:
 33%           if Ii == -1: d(t) must be bigger  than safetyD
 34%           if Ii ==  1: d(t) must be smaller than safetyD
 35%       dt = sampling interval for the collision detection
 36%       
 37%       ds = distance in s direction between two points
 38%       m = number of points wanted
 39%
 40%       kappa = curvature of the center line
 41%       y0 = distance from the center line along the y axis of the cosy of
 42%           the car
 43%       phi = angle between the x-axis of the car and the tangent on the
 44%           center line at the section between center line and y axis
 45%
 46%   OUTPUT:
 47%       flag1 = 1/0/-1
 48%           1:  everything ok
 49%           0:  solution was not unique. solution with minimal T is choosen
 50%           -1: no solution: the vectors are full of zeros
 51%       flag2 = 1/0/-1/-2
 52%           1:  everything ok
 53%           0.5: m too big, smaller one set
 54%           0:  solution was not unique. solution with minimal T is choosen
 55%           -1: the roots were imaginary
 56%           -2: no roots
 57%           -3: ds was too big
 58%       x = x in the car coordinate sytsem
 59%       y = y in the car coordinate sytsem
 60%
 61% See also
 62
 63%% init
 64x = zeros(1,m);
 65y = zeros(1,m);
 66
 67flag2 = 1;
 68
 69
 70%% coefficients
 71[flag1, ps, pd, T] = otg_pspdT_reallyDumb(S, D, kj, kT, ks, kd, dT, Tmin, Tmax, dataVeh, safetyS, safetyD, dt);
 72
 73if flag1 == -1
 74    return;
 75end
 76
 77if ((m-1)*ds > polyval(ps, T))
 78    flag2 = 0.5;
 79    m = floor(polyval(ps, T)/ds)-1;
 80    if m < 1
 81        flag2 = -3;
 82        return;
 83    end
 84end
 85
 86
 87%% calculate the times where s(t) = i*ds
 88tt = zeros(1,m);
 89
 90for k = 2:m
 91    ps_new = ps - [0 0 0 0 (k-1)*ds];
 92    r = roots(ps_new);
 93    if length(r) == 0
 94        flag2 = -2;
 95        return;
 96    elseif length(r) > 1
 97        flag2 = 0;
 98        r = r(1);
 99    end
100    
101    if ~isreal(r)
102        flag2 = -1;
103        return;
104    end
105    tt(k) = r;
106    
107end
108
109%% calc s and d at the times
110ss = polyval(ps, tt);
111dd = polyval(pd, tt);
112
113
114XY = 1/kappa*[sin(kappa*ss);1-cos(kappa*ss)] + [-sin(kappa*ss);cos(kappa*ss)].*repmat(dd,2,1);
115
116xy_car = zeros(2,m);
117
118R = [cos(phi), -sin(phi); sin(phi) cos(phi)];
119
120
121for k = 1:m
122   xy_car(:,k) =  R*XY(:,k) + [-sin(phi);cos(phi)]*(cos(phi)*y0); 
123end
124
125x = xy_car(1,:);
126y = xy_car(2,:);
127
128
129end
