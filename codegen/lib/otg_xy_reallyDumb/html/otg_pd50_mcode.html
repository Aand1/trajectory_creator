1function [pd] = otg_pd(D, T)
 2%This is the OTG = Optimal Trajectory Generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_pd returns the coefficients of the unique 5th order Polynomial for d(t)
 9%   INPUT:
10%       D = [d0, d0d, d0dd, d1] s.t.
11%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
12%       T = end time
13%
14%   OUTPUT:
15%       pd = coefficients of the polynomial in std. matlab notation i.e.
16%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
17%           + pd(6)
18%
19% See also 
20
21%init
22pd = zeros(1, 6);
23
24%precomputed formulas
25pd(1) = -(D(3)*T^2 + 6*D(2)*T + 12*D(1) - 12*D(4))/(2*T^5);
26pd(2) = (3*D(3)*T^2 + 16*D(2)*T + 30*D(1) - 30*D(4))/(2*T^4);
27pd(3) = -(3*D(3)*T^2 + 12*D(2)*T + 20*D(1) - 20*D(4))/(2*T^3);
28pd(4) = D(3)/2;
29pd(5) = D(2);
30pd(6) = D(1);
31
32
33
34end
