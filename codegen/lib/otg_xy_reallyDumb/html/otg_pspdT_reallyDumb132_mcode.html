1function [flag, ps, pd, T] = otg_pspdT_reallyDumb(S, D, kj, kT, ks, kd, dT, Tmin, Tmax, dataVeh, safetyS, safetyD, dt)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_pspdT_reallyDumb returns the coefficients of the polynomial with the
 9%best cost function and no collision
10%   INPUT:
11%       
12%       S = [v0, a0, v1] s.t.
13%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
14%       D = [d0, d0d, d0dd, d1] s.t.
15%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
16%
17%       kj = weight for the jerk functional
18%       kT = weight for the time
19%       ks = weight of the longitudinal weight function
20%       kd = weight of the lateral weight function
21%
22%       dT = time interval between two possible end Times T
23%       Tmin = minimal T
24%       Tmax = maximal T
25%
26%       dataVeh = (3xN) data of other vehivles on the road
27%           dataVeh(:,i) = [s0i; vi; Ii] s.t.
28%               s0i = initial distance between obstacle car i and own car
29%               vi  = velocity of obstacle car i anlong the road
30%               Ii  = -1/1: -1: on left lane, +1: on the right lane
31%       safetyS = min. safety distance in s direction
32%       safetyD = min. safety distance in D direction with sign:
33%           if Ii == -1: d(t) must be bigger  than safetyD
34%           if Ii ==  1: d(t) must be smaller than safetyD
35%       dt = sampling interval for the collision detection
36%
37%   OUTPUT:
38%       flag = 1/0/-1
39%           1:  everything ok
40%           0:  solution was not unique. solution with minimal T is choosen
41%           -1: no solution: the vectors are full of zeros
42%       ps = coefficients of the polynomial in std. matlab notation i.e.
43%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
44%       pd = coefficients of the polynomial in std. matlab notation i.e.
45%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
46%           + pd(6)
47%       T = end time
48%
49% See also
50
51%% init for case of return
52ps = zeros(1, 5);
53pd = zeros(1, 6);
54T = 0;
55flag = 0;
56
57%% init
58n = ceil(Tmax-Tmin)/dT; %number samples in the T space
59
60TT = linspace(Tmin, Tmax, n); %generate all Ts
61
62PS = zeros(5, n); %all the coefficients pd
63
64PD = zeros(6, n); %all the coefficients pd
65
66C = zeros(1, n); %the cost function values
67collisions = zeros(1, n); %collisions (0: no collision, 1: collision)
68
69for k = 1:n
70    PS(:, k) = otg_ps(S, TT(k)); %get coefficients
71    PD(:,k)  = otg_pd(D, TT(k)); %get coefficients
72    
73    C(k) = otg_Ctot(PS(:, k), PD(:, k), TT(k), kj, kT, ks, kd); %compute total cost function
74    
75    collisions(k) = otg_collDet_dumb(PS(:, k), PD(:, k), TT(k),dataVeh, safetyS, safetyD, dt); %check for collision
76end
77
78ind_noCol = collisions == 0; %get all with no collisions
79
80if sum(ind_noCol) == 0
81    %no trajectory with no collision
82    flag = -1;
83    return;
84end
85
86ind_sol = find((C == min(C(ind_noCol))) & ind_noCol); %the second one could be useless but there is the slight chance that a colliding path has the same C value as the noncolliding minimal one
87
88%% give solution
89ps = PS(:,ind_sol(1));
90pd = PD(:,ind_sol(1));
91T = TT(ind_sol(1));
92
93end
