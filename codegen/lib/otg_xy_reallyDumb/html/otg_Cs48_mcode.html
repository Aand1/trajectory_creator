1function [Cs] = otg_Cs(ps, T, kj, kT)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_Cs returns value of the cost function Cs = kj*Jts + kT*T for s
 9%   INPUT:
10%       ps = coefficients of the polynomial (otg_ps) s.t.
11%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
12%       T = end time
13%       kj = weight for the jerk functional
14%       kT = weight for the time
15%
16%   OUTPUT:
17%       Cs = kj*Jts + kT*T
18%
19% See also 
20
21%precomputed formula
22Cs = kj*otg_Jts(ps, T) + kT*T;
23
24end
