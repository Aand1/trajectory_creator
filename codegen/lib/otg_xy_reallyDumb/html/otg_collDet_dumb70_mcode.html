1function [collision] = otg_collDet_dumb(ps, pd, T, dataVeh, safetyS, safetyD, dt)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_collDet_dumb sees if a collision occurs (collision = 1)
 9%   INPUT:
10%       ps = coefficients of the polynomial (otg_ps) s.t.
11%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
12%       pd = coefficients of the polynomial d (otg_pd) s.t.
13%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
14%           + pd(6)
15%       T = end time
16%       dataVeh = (3xN) data of other vehivles on the road
17%           dataVeh(:,i) = [s0i; vi; Ii] s.t.
18%               s0i = initial distance between obstacle car i and own car
19%               vi  = velocity of obstacle car i anlong the road
20%               Ii  = -1/1: -1: on left lane, +1: on the right lane
21%       safetyS = min. safety distance in s direction
22%       safetyD = min. safety distance in D direction with sign:
23%           if Ii == -1: d(t) must be bigger  than safetyD
24%           if Ii ==  1: d(t) must be smaller than safetyD
25%       dt = sampling interval for the collision detection
26%
27%   OUTPUT:
28%       collsion = 0/1: 0: no collsion, 1: collsion
29%
30% See also
31
32
33N = size(dataVeh, 2); %number of obstacle vehicles
34
35tt = linspace(0, T, ceil(T/dt)); %samples
36ss = polyval(ps, tt);
37dd = polyval(pd, tt);
38
39collision = 0;
40
41for k = 1:length(tt)
42    for i = 1:N %go over all vehicles
43        
44        Ii = dataVeh(3,i);%just for readability
45        
46        if (Ii == -1) && (dd(k) < safetyD)
47            abstand = dataVeh(1,i) + dataVeh(2,i)*tt(k) - ss(k); %anfangsabstand + geschw. Auto * t - position unser Auto
48            if abstand < safetyS
49                collision = 1; %more than one collsion doesn't matter
50                return;
51            end
52        end
53        
54        if (Ii ==  1) && (dd(k) > safetyD)
55            abstand = dataVeh(1,i) + dataVeh(2,i)*tt(k) - ss(k); %anfangsabstand + geschw. Auto * t - position unser Auto
56            if abstand < safetyS
57                collision = 1; %more than one collsion doesn't matter
58                return;
59            end
60        end
61        
62        
63    end
64end
