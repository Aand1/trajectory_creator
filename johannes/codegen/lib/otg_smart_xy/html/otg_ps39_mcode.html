1function [ps] = otg_ps(S, T)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_ps returns the coefficients of the unique 4th order Polynomial for s(t)
 9%   INPUT:
10%       S = [v0, a0, v1] s.t.
11%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
12%       T = end time
13%
14%   OUTPUT:
15%       ps = coefficients of the polynomial in std. matlab notation i.e.
16%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
17%
18% See also 
19
20%init
21ps = zeros(1, 5);
22
23
24%formulas precomputed
25ps(1) = 1.0./T.^3.*(S(1).*2.0-S(3).*2.0+T.*S(2)).*(1.0./4.0);
26ps(2) = 1.0./T.^2.*(S(1).*3.0-S(3).*3.0+T.*S(2).*2.0).*(-1.0./3.0);
27ps(3) = S(2).*(1.0./2.0);
28ps(4) = S(1);
29ps(5) = 0;
30
31
32end
