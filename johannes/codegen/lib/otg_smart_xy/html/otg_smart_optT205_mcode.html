1function [flag1, flag2, flag3, Topt, TOL] = otg_smart_optT(Tstart,absTOL, maxIter, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_optT performs the optimization in T
 14%T
 15%   INPUT:
 16%
 17%       Tstart = start value for T
 18%       absTOL = (Tmax-Tmin) < absTOL --> return (the best T is guranteed
 19%           to lie within a ball with radius absTOL and middle Topt
 20%       maxIter = max. Number of Iterations (50 should be suffiecient)
 21%
 22%       S = [v0, a0, v1] s.t.
 23%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 24%       D = [d0, d0d, d0dd, d1] s.t.
 25%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 26%
 27%       kj = weight for the jerk functional
 28%       kT = weight for the time
 29%       ks = weight of the longitudinal weight function
 30%       kd = weight of the lateral weight function
 31%
 32%       dataVeh = (3x1) data of other vehivles on the road
 33%           dataVeh = [s0; v; I] s.t.
 34%               s0 = initial distance between obstacle car i and own car
 35%               v  = velocity of obstacle car i anlong the road
 36%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 37%               must be the same lane as the start of the vehicle
 38%       safetyS = min. safety distance in s direction
 39%       safetyD = min. safety distance in D direction with sign:
 40%           if I == -1: d(t) must be bigger  than safetyD
 41%           if I ==  1: d(t) must be smaller than safetyD
 42%
 43%       kappa = curvature of the circle which describes the centerline
 44%       kappaMax = max. curvature of the road in xy coordinate system
 45%       aOrthMax = max. acceleration orthogonal to the trajectory
 46%
 47%   OUTPUT:
 48%       Topt = optimal time for the trajectory
 49%       TOL = limit on the error in T
 50%       flag1 = flag for the subroutine otg_smart_objFun
 51%                1: all good
 52%               -1: the minimum velocity is negative!
 53%               -2: the safety point was not unique
 54%               -3: not considered case (by the programmer)
 55%       flag2 = flag for the subroutine otg_smart_opt_step
 56%                1: all good in this routine
 57%                0: no drivable trajectory
 58%               -1: driveability condition isn't as expected
 59%               -2: Ts not monotonically increasing
 60%       flag3 = flag for this routine
 61%                1: all good in this routine
 62%                0: no drivable trajectory
 63%               -1: number of iterations not sufficient
 64%                   -1.1: number of iterations not sufficient + sol. unique
 65%                   -1.2: number of iterations not sufficient + not unique
 66%               -2: no drivable solution was found with the given number of iterations
 67%               -10: something went wrong in one of the subroutines
 68%
 69%
 70% See also
 71
 72%% init
 73TOL = -100;
 74Topt = 0;
 75flag2 = -100;
 76flag3 = 1;
 77
 78Ts = [0.5, 1, 1.5] * Tstart;
 79[Cs, notD, coll, flag1] = otg_smart_objFun(Ts, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
 80
 81for iter = 1:maxIter
 82    % call the subroutine for one step
 83    [flag1, flag2, Ts, Cs, notD, coll] = otg_smart_opt_step(Ts, Cs, notD, coll, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
 84    
 85    if (flag1 < 0) || (flag2 <= 0)
 86        flag3 = -10;
 87        return;
 88    end
 89    
 90    if abs(Ts(3)-Ts(1)) < absTOL && (notD(2) == 0) && (coll(2) == 0)
 91        flag3 = 1;
 92        Topt = Ts(2);
 93        TOL = Ts(3) - Ts(1);
 94        return;
 95    end
 96    
 97    
 98    
 99end
100
101%% number of iterations was not sufficient
102indOk = (notD == 0) & (coll == 0);
103
104if sum(indOk) > 0
105    %at least one drivable path
106    if sum(indOk) == 1
107        %the drivable one is unique
108        flag3 = -1.1;
109        Topt = Ts(indOk);
110        TOL = Ts(3) - Ts(1);
111        return
112    end
113    
114    % the drivable path is not unique, just choose the first one
115    flag3 = -1.1;
116    indOk = find(indOk);
117    Topt = Ts(indOk(1));
118    TOL = Ts(3) - Ts(1);
119    return;
120end
121
122%no drivable solution was found with the given number of iterations
123flag3 = -2;
124return
125
126
127end
