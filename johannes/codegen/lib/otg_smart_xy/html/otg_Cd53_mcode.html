1function [Cd] = otg_Cd(pd, T, kj, kT)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_Cd returns value of the cost function Cd = kj*Jtd + kT*T for d
 9%   INPUT:
10%       pd = coefficients of the polynomial d (otg_pd) s.t.
11%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
12%           + pd(6)
13%       T = end time
14%       kj = weight for the jerk functional
15%       kT = weight for the time
16%
17%   OUTPUT:
18%       Cd = kj*Jtd + kT*T
19%
20% See also 
21
22%precomputed formula
23Cd = kj*otg_Jtd(pd, T) + kT*T;
24end
