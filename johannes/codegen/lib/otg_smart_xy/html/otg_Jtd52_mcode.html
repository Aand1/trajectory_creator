1function [Jtd] = otg_Jtd(pd, T)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_Jtd returns value of the functional Jt = int_0^T  (d'''(t))^2 dt for d
 9%   INPUT:
10%       pd = coefficients of the polynomial d (otg_pd) s.t.
11%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
12%           + pd(6)
13%       T = end time
14%
15%   OUTPUT:
16%       Jtd = int_0^T  (d'''(t))^2 dt
17%
18% See also 
19
20%precomputed formula
21Jtd = 36*T*pd(3)^2 + T^3*(192*pd(2)^2 + 240*pd(3)*pd(1)) + 720*T^5*pd(1)^2 + 144*T^2*pd(3)*pd(2) + 720*T^4*pd(2)*pd(1);
22
23end
