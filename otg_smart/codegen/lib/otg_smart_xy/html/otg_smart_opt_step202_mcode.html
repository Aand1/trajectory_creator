1function [flag1, flag2, Ts_new, Cs_new, notD_new, coll_new] = otg_smart_opt_step(Ts, Cs, notD, coll, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_opt_step performs one step of the iterative search for the best
 14%T
 15%   INPUT:
 16%
 17%       Ts = [1x3] 3 time points of the current optim step
 18%       Cs = [1x3] 3 values of the cost function belonging to the Ts
 19%       notD = [1x3] 3 values of drivability cond. belonging to the Ts
 20%           = 0 <==> trajectory i is drivable
 21%           = 1 <==> trajectory i not drivable because of curvature
 22%           = 2 <==> trajectory i not drivable because of orth. acceleration
 23%           = 3 <==> trajectory i not drivable because of both above
 24%       coll = [1x3] 3 values of collision cond. belonging to the Ts
 25%           = 1 <==> trajectory i is colliding
 26%
 27%       S = [v0, a0, v1] s.t.
 28%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 29%       D = [d0, d0d, d0dd, d1] s.t.
 30%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 31%
 32%       kj = weight for the jerk functional
 33%       kT = weight for the time
 34%       ks = weight of the longitudinal weight function
 35%       kd = weight of the lateral weight function
 36%
 37%       dataVeh = (3x1) data of other vehivles on the road
 38%           dataVeh = [s0; v; I] s.t.
 39%               s0 = initial distance between obstacle car i and own car
 40%               v  = velocity of obstacle car i anlong the road
 41%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 42%               must be the same lane as the start of the vehicle
 43%       safetyS = min. safety distance in s direction
 44%       safetyD = min. safety distance in D direction with sign:
 45%           if I == -1: d(t) must be bigger  than safetyD
 46%           if I ==  1: d(t) must be smaller than safetyD
 47%
 48%       kappa = curvature of the circle which describes the centerline
 49%       kappaMax = max. curvature of the road in xy coordinate system
 50%       aOrthMax = max. acceleration orthogonal to the trajectory
 51%
 52%   OUTPUT:
 53%       Ts_new = [1x3] 3 time points of the next optim step
 54%       Cs_new = [1x3] 3 values of the cost function belonging to the Ts_new
 55%       notD_new = [1x3] 3 values of drivability cond. belonging to the
 56%               Ts_new
 57%           = 0 <==> trajectory i is drivable
 58%           = 1 <==> trajectory i not drivable because of curvature
 59%           = 2 <==> trajectory i not drivable because of orth. acceleration
 60%           = 3 <==> trajectory i not drivable because of both above
 61%       coll_new = [1x3] 3 values of collision cond. belonging to the
 62%               Ts_new
 63%           = 1 <==> trajectory i is colliding
 64%       flag1 = flag for the subroutine otg_smart_objFun 
 65%                1: all good
 66%               -1: the minimum velocity is negative!
 67%               -2: the safety point was not unique
 68%               -3: not considered case (by the programmer)
 69%       flag2 = flag for this routine
 70%                1: all good in this routine
 71%                0: no drivable trajectory
 72%               -1: driveability condition isn't as expected
 73%               -2: Ts not monotonically increasing 
 74%
 75% See also
 76
 77%% init
 78
 79flag2 = 1;
 80Ts_new = zeros(1,3);
 81Cs_new = zeros(1,3);
 82notD_new = zeros(1,3);
 83coll_new = zeros(1,3);
 84flag1 = 0;
 85
 86%% check for order
 87if ~((Ts(1) < Ts(2))&&(Ts(2) < Ts(3)))
 88    flag2 = -2;
 89    return;
 90end
 91
 92%% check if there is an error with the drivability
 93if (notD(3) > 0)
 94    %all points should be non drivable
 95    if (notD(1) == 0) || (notD(2) == 0)
 96        %something went wrong about the drivability
 97        flag2 = -1;
 98        return;
 99    end
100end
101
102%% check if there is an error with the collisions
103if (coll(1) > 0)
104    %all points should be colliding
105    if (coll(2) == 0) || (coll(3) == 0)
106        %something went wrong about the collsion
107        flag2 = -3;
108        return;
109    end
110end
111
112%% check if no valid trajectory exists
113if sum((notD > 0) & (coll > 0)) > 0
114    %there is at least one traj wich is both not drivable and not colliding
115    %--> no valid trajectory exists
116    flag2 = 0;
117    return
118end
119
120
121%% all non drivable
122if (notD(1) > 0) && (notD(2) > 0) && (notD(3) > 0)
123    Ts_new(1) = Ts(2);
124    Ts_new(2) = Ts(3);
125    Ts_new(3) = Ts(3) + (Ts(3)-Ts(1));
126    
127    [C3, notD3, coll3, flag1] = otg_smart_objFun(Ts_new(3), S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
128    
129    if flag1 < 0
130        %something in the subroutine went wrong
131        return;
132    end
133    
134    Cs_new(1) = Cs(2);
135    Cs_new(2) = Cs(3);
136    Cs_new(3) = C3;
137    
138    notD_new(1) = notD(2);
139    notD_new(2) = notD(3);
140    notD_new(3) = notD3;
141    
142    coll_new(1) = coll(2);
143    coll_new(2) = coll(3);
144    coll_new(3) = coll3;
145    
146    return;
147end
148
149%% all collidiong
150if (coll(1) > 0) && (coll(2) > 0) && (coll(3) > 0)
151    Ts_new(1) = Ts(1) - (Ts(3) - Ts(1)); %fibonacci growth
152    Ts_new(2) = Ts(1);
153    Ts_new(3) = Ts(2);
154    
155    [C1, notD1, coll1, flag1] = otg_smart_objFun(Ts_new(1), S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
156    
157    if flag1 < 0
158        %something in the subroutine went wrong
159        return;
160    end
161    
162    Cs_new(1) = C1;
163    Cs_new(2) = Cs(1);
164    Cs_new(3) = Cs(2);
165    
166    notD_new(1) = notD1;
167    notD_new(2) = notD(1);
168    notD_new(3) = notD(2);
169    
170    coll_new(1) = coll1;
171    coll_new(2) = coll(1);
172    coll_new(3) = coll(2);
173    
174    return;
175end
176
177%% Test all the cases
178%workaround for cpp not offering a value inf: if one trajectory is not
179%drivable or colliding set its c value to double the max
180
181maxC = max(Cs);
182
183Cs((notD>0) | (coll>0)) = maxC;
184
185if (Cs(1) < Cs(2)) && (Cs(2) <= Cs(3))
186    
187    Ts_new(1) = Ts(1) - (Ts(3) - Ts(1)); %fibonacci growth
188    Ts_new(2) = Ts(1);
189    Ts_new(3) = Ts(2);
190    
191    [C1, notD1, coll1, flag1] = otg_smart_objFun(Ts_new(1), S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
192    
193    if flag1 < 0
194        %something in the subroutine went wrong
195        return;
196    end
197    
198    Cs_new(1) = C1;
199    Cs_new(2) = Cs(1);
200    Cs_new(3) = Cs(2);
201    
202    notD_new(1) = notD1;
203    notD_new(2) = notD(1);
204    notD_new(3) = notD(2);
205    
206    coll_new(1) = coll1;
207    coll_new(2) = coll(1);
208    coll_new(3) = coll(2);
209    
210    return;
211end
212
213if (Cs(1) >= Cs(2)) && (Cs(2) > Cs(3))
214    Ts_new(1) = Ts(2);
215    Ts_new(2) = Ts(3);
216    Ts_new(3) = Ts(3) + (Ts(3)-Ts(1));%fibonacci growth
217    
218    [C3, notD3, coll3, flag1] = otg_smart_objFun(Ts_new(3), S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
219    
220    if flag1 < 0
221        %something in the subroutine went wrong
222        return;
223    end
224    
225    Cs_new(1) = Cs(2);
226    Cs_new(2) = Cs(3);
227    Cs_new(3) = C3;
228    
229    notD_new(1) = notD(2);
230    notD_new(2) = notD(3);
231    notD_new(3) = notD3;
232    
233    coll_new(1) = coll(2);
234    coll_new(2) = coll(3);
235    coll_new(3) = coll3;
236    
237    return;
238end
239
240if (Cs(1) >= Cs(2)) && (Cs(2) <= Cs(3))
241    
242    Ta = (Ts(1)+Ts(2))/2;
243    Tb = (Ts(2)+Ts(3))/2;
244    
245    [Ca, notDa, colla, flag1] = otg_smart_objFun(Ta, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
246    if flag1 < 0
247        %something in the subroutine went wrong
248        return;
249    end
250    
251    [Cb, notDb, collb, flag1] = otg_smart_objFun(Tb, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
252    if flag1 < 0
253        %something in the subroutine went wrong
254        return;
255    end
256    
257    if (notDa > 0) || (colla > 0)
258        Ca = maxC;
259    end
260    
261    if (notDb > 0) || (collb > 0)
262        Cb = maxC;
263    end
264    
265    if (Cb < Cs(2))
266        
267        %out of the five points the minumum must be somewhere between the
268        %three outter right ones
269        
270        Ts_new(1) = Ts(2);
271        Ts_new(2) = Tb;
272        Ts_new(3) = Ts(3);
273        
274        Cs_new(1) = Cs(2);
275        Cs_new(2) = Cb;
276        Cs_new(3) = Cs(3);
277        
278        notD_new(1) = notD(2);
279        notD_new(2) = notDb;
280        notD_new(3) = notD(3);
281        
282        coll_new(1) = coll(2);
283        coll_new(2) = collb;
284        coll_new(3) = coll(3);
285        
286        return;
287    end
288    
289    if (Ca < Cs(2))
290        
291        %out of the five points the minumum must be somewhere between the
292        %three outter left ones
293        
294        Ts_new(1) = Ts(1);
295        Ts_new(2) = Ta;
296        Ts_new(3) = Ts(2);
297        
298        Cs_new(1) = Cs(1);
299        Cs_new(2) = Ca;
300        Cs_new(3) = Cs(2);
301        
302        notD_new(1) = notD(1);
303        notD_new(2) = notDa;
304        notD_new(3) = notD(2);
305        
306        coll_new(1) = coll(1);
307        coll_new(2) = colla;
308        coll_new(3) = coll(2);
309        
310        return;
311    end
312    
313    if (Ca >= Cs(2)) && (Cb >= Cs(2))
314        
315        %out of the five points the minumum must be somewhere between the
316        %three middle ones
317        
318        Ts_new(1) = Ta;
319        Ts_new(2) = Ts(2);
320        Ts_new(3) = Tb;
321        
322        Cs_new(1) = Ca;
323        Cs_new(2) = Cs(2);
324        Cs_new(3) = Cb;
325        
326        notD_new(1) = notDa;
327        notD_new(2) = notD(2);
328        notD_new(3) = notDb;
329        
330        coll_new(1) = colla;
331        coll_new(2) = coll(2);
332        coll_new(3) = collb;
333        
334        return;
335    end
336    
337end
