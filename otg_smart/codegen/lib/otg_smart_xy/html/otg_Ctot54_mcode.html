1function [Ctot] = otg_Ctot(ps, pd, T, kj, kT, ks, kd)
 2%This is the OTG = Optimal trajectory generation package. This is based on
 3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
 4%Frenet Frame"(2010) by Werling et. al.
 5%
 6%--------------------------------------------------------------------------
 7%
 8%otg_Ctot returns value of the cost total function Ctot = ks*Cs + kd*Cd
 9%   INPUT:
10%       ps = coefficients of the polynomial (otg_ps) s.t.
11%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
12%       pd = coefficients of the polynomial d (otg_pd) s.t.
13%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
14%           + pd(6)
15%       T = end time
16%       kj = weight for the jerk functional
17%       kT = weight for the time
18%       ks = weight of the longitudinal weight function
19%       kd = weight of the lateral weight function
20%
21%   OUTPUT:
22%       Ctot = ks*Cs + kd*Cd
23%
24% See also 
25
26Ctot = ks*otg_Cs(ps, T, kj, kT) + kd*otg_Cd(pd, T, kj, kT);
27end
