1function [flag1, flag2, flag3, flagAll, x, y, T, TOL] = otg_smart_xy(absTOL, maxIter, v1, d1, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappaMax, aOrthMax, m, kappa, y0, phi, vx0, ax0, w)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_opt_xy gives the coordinates x and y of the optimal trjectory in
 14%the car coordinate system
 15%   INPUT:
 16%
 17%       absTOL = (Tmax-Tmin) < absTOL --> return (the best T is guranteed
 18%           to lie within a ball with radius absTOL and middle Topt
 19%       maxIter = max. Number of Iterations (50 should be suffiecient)
 20%
 21%
 22%       v1 = velocity in s direction at the end of the trajectory
 23%       d1 = distance from the center line at the end of the trajectory
 24%
 25%       kj = weight for the jerk functional
 26%       kT = weight for the time
 27%       ks = weight of the longitudinal weight function
 28%       kd = weight of the lateral weight function
 29%
 30%       dataVeh = (3x1) data of other vehivles on the road
 31%           dataVeh = [s0; v; I] s.t.
 32%               s0 = initial distance between obstacle car i and own car
 33%               v  = velocity of obstacle car i anlong the road
 34%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 35%               must be the same lane as the start of the vehicle
 36%       safetyS = min. safety distance in s direction
 37%       safetyD = min. safety distance in D direction with sign:
 38%           if I == -1: d(t) must be bigger  than safetyD
 39%           if I ==  1: d(t) must be smaller than safetyD
 40%
 41%       kappaMax = max. curvature of the road in xy coordinate system
 42%       aOrthMax = max. acceleration orthogonal to the trajectory
 43%
 44%       m = number of points wanted
 45%
 46%       kappa = curvature of the center line
 47%       phi = angle between the x-axis of the car and the tangent on the
 48%           center line at the section between center line and y axis
 49%
 50%   OUTPUT:
 51%       x = x in the car coordinate sytsem
 52%       y = y in the car coordinate sytsem
 53%       T = end time
 54%       TOL = limit on the error in T
 55%
 56%       flag1 = flag for the subroutine otg_smart_objFun
 57%                1: all good
 58%               -1: the minimum velocity is negative!
 59%               -2: the safety point was not unique
 60%               -3: not considered case (by the programmer)
 61%       flag2 = flag for the subroutine otg_smart_opt_step
 62%                1: all good in this routine
 63%                0: no drivable trajectory
 64%               -1: driveability condition isn't as expected
 65%               -2: Ts not monotonically increasing
 66%       flag3 = flag for the otg_smart_optT subroutine
 67%                1: all good in this routine
 68%                0: no drivable trajectory
 69%               -1: number of iterations not sufficient
 70%                   -1.1: number of iterations not sufficient + sol. unique
 71%                   -1.2: number of iterations not sufficient + not unique
 72%               -2: no drivable solution was found with the given number of iterations
 73%               -10: something went wrong in one of the subroutines
 74%       flagAll = all together
 75%               1: all is good;
 76%               -1: something went somwhere wrong
 77%
 78%
 79% See also
 80
 81%% init
 82x = zeros(1,m);
 83y = zeros(1,m);
 84
 85%% converte the state to S, D
 86S = [cos(phi)*vx0,-sin(phi)*w*vx0+cos(phi)*ax0, v1]; %really dirty
 87
 88cosphi_d = -sin(phi)*w;
 89cosphi_dd = -w^2*cos(phi);
 90y0_d = -sin(phi)*vx0;
 91y0_dd = -(cos(phi)*w*vx0+sin(phi)*ax0);
 92
 93D = [-cos(phi)*y0,-(cosphi_d*y0+cos(phi)*y0_d),-(cosphi_dd*y0+2*cosphi_d*y0_d+cos(phi)*y0_dd),d1];
 94
 95
 96%% coefficients
 97[flag1, flag2, flag3, flagAll, ps, pd, T, TOL] = otg_smart_pspdT(absTOL, maxIter, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
 98
 99if flagAll == -1
100    return;
101end
102
103
104%% calculate tt
105tt = linspace(0,T(1),m);
106
107%% calc s and d at the times
108ss = polyval(ps, tt);
109dd = polyval(pd, tt);
110
111if kappa >0
112    XY = 1/kappa*[sin(kappa*ss);1-cos(kappa*ss)] + [-sin(kappa*ss);cos(kappa*ss)].*repmat(dd,2,1);
113else
114    XY = [ss;zeros(1, m)] + repmat([0;1],1,m).*repmat(dd,2,1);
115end
116
117xy_car = zeros(2,m);
118
119R = [cos(phi), -sin(phi); sin(phi) cos(phi)];
120
121
122for k = 1:m
123    xy_car(:,k) =  R*XY(:,k) + [-sin(phi);cos(phi)]*(-D(1));
124end
125
126x = xy_car(1,:);
127y = xy_car(2,:);
128
129
130
131
132
133end
