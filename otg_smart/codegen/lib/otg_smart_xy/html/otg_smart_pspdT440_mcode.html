1function [flag1, flag2, flag3, flagAll, ps, pd, T, TOL] = otg_smart_pspdT(absTOL, maxIter, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_opt_pspdT gives ps, pd, T for the optimal trajectory
 14%T
 15%   INPUT:
 16%
 17%       absTOL = (Tmax-Tmin) < absTOL --> return (the best T is guranteed
 18%           to lie within a ball with radius absTOL and middle Topt
 19%       maxIter = max. Number of Iterations (50 should be suffiecient)
 20%
 21%
 22%       S = [v0, a0, v1] s.t.
 23%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 24%       D = [d0, d0d, d0dd, d1] s.t.
 25%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 26%
 27%       kj = weight for the jerk functional
 28%       kT = weight for the time
 29%       ks = weight of the longitudinal weight function
 30%       kd = weight of the lateral weight function
 31%
 32%       dataVeh = (3x1) data of other vehivles on the road
 33%           dataVeh = [s0; v; I] s.t.
 34%               s0 = initial distance between obstacle car i and own car
 35%               v  = velocity of obstacle car i anlong the road
 36%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 37%               must be the same lane as the start of the vehicle
 38%       safetyS = min. safety distance in s direction
 39%       safetyD = min. safety distance in D direction with sign:
 40%           if I == -1: d(t) must be bigger  than safetyD
 41%           if I ==  1: d(t) must be smaller than safetyD
 42%
 43%       kappa = curvature of the circle which describes the centerline
 44%       kappaMax = max. curvature of the road in xy coordinate system
 45%       aOrthMax = max. acceleration orthogonal to the trajectory
 46%
 47%   OUTPUT:
 48%       ps = coefficients of the polynomial in std. matlab notation i.e.
 49%           s(t) = ps(1)*t^4 + ps(2)*t^3 + ps(3)*t^2 + ps(4)*t + ps(5)
 50%       pd = coefficients of the polynomial in std. matlab notation i.e.
 51%           d(t) = pd(1)*t^5 + pd(2)*t^4 + pd(3)*t^3 + pd(4)*t^2 + pd(5)*t
 52%           + pd(6)
 53%       T = end time
 54%       TOL = limit on the error in T
 55%
 56%       flag1 = flag for the subroutine otg_smart_objFun
 57%                1: all good
 58%               -1: the minimum velocity is negative!
 59%               -2: the safety point was not unique
 60%               -3: not considered case (by the programmer)
 61%       flag2 = flag for the subroutine otg_smart_opt_step
 62%                1: all good in this routine
 63%                0: no drivable trajectory
 64%               -1: driveability condition isn't as expected
 65%               -2: Ts not monotonically increasing
 66%       flag3 = flag for the otg_smart_optT subroutine
 67%                1: all good in this routine
 68%                0: no drivable trajectory
 69%               -1: number of iterations not sufficient
 70%                   -1.1: number of iterations not sufficient + sol. unique
 71%                   -1.2: number of iterations not sufficient + not unique
 72%               -2: no drivable solution was found with the given number of iterations
 73%               -10: something went wrong in one of the subroutines
 74%       flagAll = all together
 75%               1: all is good;
 76%               -1: something went somwhere wrong
 77%
 78%
 79% See also
 80
 81%% init
 82ps = zeros(5,1);
 83pd = zeros(6,1);
 84
 85flagAll = -1;
 86
 87Tstart = dataVeh(1) / (abs((S(1)+S(3))/2 - 0.5*dataVeh(2)) + 0.001);
 88
 89if (Tstart <= 1) || (Tstart >= 20)
 90    Tstart = 5; %just some hard-coded plausibility check
 91end
 92
 93%% get the best T
 94[flag1, flag2, flag3, T, TOL] = otg_smart_optT(Tstart,absTOL, maxIter, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
 95
 96if (flag1 < 0) || (flag2 <= 0) || (flag3 <= 0)
 97    return; %if no return: all good
 98end
 99
100% get the coefficients
101ps = otg_ps(S,T);
102pd = otg_pd(D,T);
103
104% see if not colliding
105%too lazy, should be ok
106[C, notD, coll, ~] = otg_smart_objFun(T, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax);
107
108if (notD(1) == 0) && (coll(1) == 0)
109    flagAll = 1;
110end
111
112
113
114
115
116
117
118end
