1function [Ctot, notD, coll, flag] = otg_smart_objFun(T, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_objFun returns for given T and parameters the value of the cost
 14%function C_tot and indicators if the trajectory corresponding to T is
 15%drivable
 16%   INPUT:
 17%
 18%       T = [1xn] for n different end times T(i)
 19%
 20%       S = [v0, a0, v1] s.t.
 21%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 22%       D = [d0, d0d, d0dd, d1] s.t.
 23%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 24%
 25%       kj = weight for the jerk functional
 26%       kT = weight for the time
 27%       ks = weight of the longitudinal weight function
 28%       kd = weight of the lateral weight function
 29%
 30%       dataVeh = (3x1) data of other vehivles on the road
 31%           dataVeh = [s0; v; I] s.t.
 32%               s0 = initial distance between obstacle car i and own car
 33%               v  = velocity of obstacle car i anlong the road
 34%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 35%               must be the same lane as the start of the vehicle
 36%       safetyS = min. safety distance in s direction
 37%       safetyD = min. safety distance in D direction with sign:
 38%           if I == -1: d(t) must be bigger  than safetyD
 39%           if I ==  1: d(t) must be smaller than safetyD
 40%
 41%       kappa = curvature of the circle which describes the centerline
 42%       kappaMax = max. curvature of the road in xy coordinate system
 43%       aOrthMax = max. acceleration orthogonal to the trajectory
 44%
 45%   OUTPUT:
 46%       C_tot = [1xn]: cost function values for the different T(i)
 47%       notD = [1xn] = 0/1/2/3 indicator if the trajectory is not drivable: notD(i)
 48%           = 0 <==> trajectory i is drivable
 49%           = 1 <==> trajectory i not drivable because of curvature
 50%           = 2 <==> trajectory i not drivable because of orth. acceleration
 51%           = 3 <==> trajectory i not drivable because of both above
 52%       coll = [1xn]: indicator if the trajectory is colliding: coll(i)
 53%           = 1 <==> trajectory i is colliding
 54%       flag
 55%           1: all good
 56%           -1: the minimum velocity is negative!
 57%           -2: the safety point was not unique
 58%           -3: not considered case (by the programmer)
 59%
 60% See also
 61
 62%% init
 63n = length(T);
 64flag = 1;
 65
 66Ctot = zeros(1,n);
 67notD = zeros(1,n);
 68coll = zeros(1,n);
 69
 70
 71%% Get the values of ps, pd for the Ts
 72PS = zeros(5,n);
 73PD = zeros(6,n);
 74
 75for i = 1:n
 76    PS(:,i) = otg_ps(S,T(i));
 77    PD(:,i) = otg_pd(D,T(i));
 78end
 79
 80%% Calculate the costs
 81
 82for i = 1:n
 83    Ctot(i) = otg_Ctot(PS(:,i),PD(:,i),T(i), kj, kT, ks, kd);
 84end
 85
 86
 87%% See if drivable: We use a very rough approximation here!!
 88
 89for i = 1:n
 90    %LENKEINSCHLAG
 91    d_d = polyder(PD(:,i)); %first derivative w.r.t. t of d(t)
 92    d_dd = polyder(d_d); %second derivative w.r.t. t of d(t)
 93    d_ddd = polyder(d_dd); %third derivative to find the max of the second derivative
 94    r = roots(d_ddd)';%get all roots of THIRD derivative
 95    
 96    ind = imag(r) == 0;%only real roots
 97    r = r(ind);
 98    
 99    ind = (r>0) & (r < T(i)); %only roots in interval
100    r = r(ind);
101    
102    ts_d = [0 r T(i)]; %also max at the border
103    d_dd_ts = polyval(d_dd, ts_d); %HERE ONE NEEDS THE SECOND DERIVATIVE
104    
105    d_dd_max = max(d_dd_ts); %get the maximal value of the polynomial on the interval
106    
107    
108    s_d = polyder(PS(:,i)); %first derivative w.r.t. t of s(t) = v(t)
109    s_dd = polyder(s_d); %second derivative to find the
110    r = roots(s_dd)';%get all roots of first derivative
111    
112    ind = imag(r) == 0;%only real roots
113    r = r(ind);
114    
115    ind = (r>0) & (r < T(i)); %only roots in interval
116    r = r(ind);
117    
118    ts_s = [0 r T(i)]; %also min at the border
119    s_d_ts = polyval(s_d, ts_s);
120    
121    s_d_min = min(s_d_ts); %get the min value of the polynomial on the interval
122    s_d_max = max(s_d_ts); %get the maximal value of the velocity in s direction
123    
124    if (s_d_min) <= 0
125        %something went terrible wrong. we are driving backwards
126        flag = -1;
127        return;
128    end
129    
130    %now the bound for the curvature can be calculated
131    kappa_xy_max = abs(kappa) + d_dd_max/s_d_min;
132    
133    if kappa_xy_max > kappaMax
134        %the curvature could be (according to our approximation) to big
135        notD(i) = 1;
136    end
137    
138    
139    %MAXIMALE ORTH: BESCHLEUNIGUNG
140    r = roots(d_dd)';%get all roots of SECOND derivative
141    
142    ind = imag(r) == 0;%only real roots
143    r = r(ind);
144    
145    ind = (r>0) & (r < T(i)); %only roots in interval
146    r = r(ind);
147    
148    ts_d_2 = [0 r T(i)]; %also max at the border
149    d_d_ts = polyval(d_d, ts_d_2); %HERE ONE NEEDS THE SECOND DERIVATIVE
150    
151    d_d_max = max(d_d_ts); %get the maximal value of the polynomial on the interval
152    
153    vx_max = sqrt(s_d_max^2+d_d_max^2);
154    
155    if vx_max^2*kappa_xy_max > aOrthMax
156        %the orthogonal acceleration is too much for the grip
157        notD(i) = notD(i)+2; %must be 2. 2+1 = 3 (all), 0 + 2= 2 (only acc.)
158    end
159end
160
161
162
163%% check for collision
164for i = 1:n
165    %s(t) should be an increasing function. This is guranteed because we
166    %calculate the min of the 1st derivative above and throw an error if
167    %this is non positive
168    
169    abstandMinusSafety_poly = [0 0 0 dataVeh(2) dataVeh(1)-safetyS]' - PS(:,i);
170    
171    r = roots(abstandMinusSafety_poly);
172    
173    ind = imag(r) == 0;%only real roots
174    r = r(ind);
175    
176    ind = (r>0) & (r < T(i)); %only roots in interval
177    r = r(ind);
178    
179    if (isempty(r)) && (polyval(abstandMinusSafety_poly, T(i)/2) > 0)
180        %all good no collsion possible
181    elseif (isempty(r)) && (polyval(abstandMinusSafety_poly, T(i)/2) <= 0)
182        %fix collsion
183        coll(i) = 1;
184    elseif (length(r) == 1) && (polyval(abstandMinusSafety_poly, r(1)/2) > 0)
185        %one unique possible collision possible
186        d_r = polyval(PD(:,i),r(1));
187        
188        I = dataVeh(3);
189        
190        if (I == -1) && (d_r < safetyD)
191            coll(i) = 1;
192        end
193        
194        if (I ==  1) && (d_r > safetyD)
195            coll(i) = 1;
196        end
197    elseif (length(r) == 1) && (polyval(abstandMinusSafety_poly, r(1)/2) > 0)
198        %there is a collision directly at the start
199        coll(i) = -1;
200    else
201        flag = -3;
202        return;
203    end
204end
