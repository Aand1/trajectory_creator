1function [Ctot, notD, coll, flag] = otg_smart_objFun(T, S, D, kj, kT, ks, kd, dataVeh, safetyS, safetyD, kappa, kappaMax, aOrthMax)
  2%This is the OTG = Optimal trajectory generation package. This is based on
  3%the paper "Optimal Trajectory Generation for Dynamic Street Scenarios in a
  4%Frenet Frame"(2010) by Werling et. al.
  5%
  6%This is the otg_smart subpackage which relies on the assumption that there
  7%is only one other vehicle on the road. This improves the optimation
  8%drastically in speed and accuracy. If this fails the otg_dumb subpackage
  9%should be used
 10%
 11%--------------------------------------------------------------------------
 12%
 13%otg_smart_objFun returns for given T and parameters the value of the cost
 14%function C_tot and indicators if the trajectory corresponding to T is
 15%drivable
 16%   INPUT:
 17%
 18%       T = [1xn] for n different end times T(i)
 19%
 20%       S = [v0, a0, v1] s.t.
 21%           s(0) = 0, s'(0) = v0, s''(0) = a0, s'(T) = v1
 22%       D = [d0, d0d, d0dd, d1] s.t.
 23%           d(0) = d0, d'(0) = d0d, d''(0) = d0dd, d'(T) = d1
 24%
 25%       kj = weight for the jerk functional
 26%       kT = weight for the time
 27%       ks = weight of the longitudinal weight function
 28%       kd = weight of the lateral weight function
 29%
 30%       dataVeh = (3x1) data of other vehivles on the road
 31%           dataVeh = [s0; v; I] s.t.
 32%               s0 = initial distance between obstacle car i and own car
 33%               v  = velocity of obstacle car i anlong the road
 34%               I  = -1/1: -1: on right lane, +1: on the left lane: this
 35%               must be the same lane as the start of the vehicle
 36%       safetyS = min. safety distance in s direction
 37%       safetyD = min. safety distance in D direction with sign:
 38%           if I == -1: d(t) must be bigger  than safetyD
 39%           if I ==  1: d(t) must be smaller than safetyD
 40%
 41%       kappa = curvature of the circle which describes the centerline
 42%       kappaMax = max. curvature of the road in xy coordinate system
 43%       aOrthMax = max. acceleration orthogonal to the trajectory
 44%
 45%   OUTPUT:
 46%       C_tot = [1xn]: cost function values for the different T(i)
 47%       notD = [1xn] = 0/1/2/3 indicator if the trajectory is not drivable: notD(i)
 48%           = 0 <==> trajectory i is drivable
 49%           = 1 <==> trajectory i not drivable because of curvature
 50%           = 2 <==> trajectory i not drivable because of orth. acceleration
 51%           = 3 <==> trajectory i not drivable because of both above
 52%       coll = [1xn]: indicator if the trajectory is colliding: coll(i)
 53%           = 1 <==> trajectory i is colliding
 54%       flag
 55%           1: all good
 56%           -1: the minimum velocity is negative!
 57%           -2: the safety point was not unique
 58%           -3: not considered case (by the programmer)
 59%
 60% See also
 61
 62%% init
 63nSamples = 100;
 64
 65polyrootfind = 0;
 66
 67n = length(T);
 68flag = 1;
 69
 70Ctot = zeros(1,n);
 71notD = zeros(1,n);
 72coll = zeros(1,n);
 73
 74
 75%% Get the values of ps, pd for the Ts
 76PS = zeros(5,n);
 77PD = zeros(6,n);
 78
 79for i = 1:n
 80    PS(:,i) = otg_ps(S,T(i));
 81    PD(:,i) = otg_pd(D,T(i));
 82end
 83
 84%% Calculate the costs
 85
 86for i = 1:n
 87    Ctot(i) = otg_Ctot(PS(:,i),PD(:,i),T(i), kj, kT, ks, kd);
 88end
 89
 90
 91%% See if drivable: We use a very rough approximation here!!
 92
 93for i = 1:n
 94    % polynomial root finding
 95    
 96    if polyrootfind
 97        %LENKEINSCHLAG
 98        d_d = polyder(PD(:,i)); %first derivative w.r.t. t of d(t)
 99        d_dd = polyder(d_d); %second derivative w.r.t. t of d(t)
100        d_ddd = polyder(d_dd); %third derivative to find the max of the second derivative
101        r = roots(d_ddd)';%get all roots of THIRD derivative
102        
103        ind = imag(r) == 0;%only real roots
104        r = r(ind);
105        
106        ind = (r>0) & (r < T(i)); %only roots in interval
107        r = r(ind);
108        
109        ts_d = [0 r T(i)]; %also max at the border
110        d_dd_ts = polyval(d_dd, ts_d); %HERE ONE NEEDS THE SECOND DERIVATIVE
111        
112        d_dd_max = max(d_dd_ts); %get the maximal value of the polynomial on the interval
113        
114        
115        s_d = polyder(PS(:,i)); %first derivative w.r.t. t of s(t) = v(t)
116        s_dd = polyder(s_d); %second derivative to find the
117        r = roots(s_dd)';%get all roots of first derivative
118        
119        ind = imag(r) == 0;%only real roots
120        r = r(ind);
121        
122        ind = (r>0) & (r < T(i)); %only roots in interval
123        r = r(ind);
124        
125        ts_s = [0 r T(i)]; %also min at the border
126        s_d_ts = polyval(s_d, ts_s);
127        
128        s_d_min = min(s_d_ts); %get the min value of the polynomial on the interval
129        s_d_max = max(s_d_ts); %get the maximal value of the velocity in s direction
130        
131        if (s_d_min) <= 0
132            %something went terrible wrong. we are driving backwards
133            flag = -1;
134            return;
135        end
136        
137        %now the bound for the curvature can be calculated
138        kappa_xy_max = abs(kappa) + d_dd_max/s_d_min;
139        
140        if kappa_xy_max > kappaMax
141            %the curvature could be (according to our approximation) to big
142            notD(i) = 1;
143        end
144        
145        
146        %MAXIMALE ORTH: BESCHLEUNIGUNG
147        r = roots(d_dd)';%get all roots of SECOND derivative
148        
149        ind = imag(r) == 0;%only real roots
150        r = r(ind);
151        
152        ind = (r>0) & (r < T(i)); %only roots in interval
153        r = r(ind);
154        
155        ts_d_2 = [0 r T(i)]; %also max at the border
156        d_d_ts = polyval(d_d, ts_d_2); %HERE ONE NEEDS THE SECOND DERIVATIVE
157        
158        d_d_max = max(d_d_ts); %get the maximal value of the polynomial on the interval
159        
160        vx_max = sqrt(s_d_max^2+d_d_max^2);
161        
162        if vx_max^2*kappa_xy_max > aOrthMax
163            %the orthogonal acceleration is too much for the grip
164            notD(i) = notD(i)+2; %must be 2. 2+1 = 3 (all), 0 + 2= 2 (only acc.)
165        end
166        
167    end
168    
169    
170    %sampling based approach
171    if ~polyrootfind
172        tt = linspace(0, T(i), nSamples);
173        
174        d_d = polyder(PD(:,i)); %first derivative w.r.t. t of d(t)
175        d_dd = polyder(d_d); %second derivative w.r.t. t of d(t)
176        
177        s_d = polyder(PS(:,i)); %first derivative w.r.t. t of s(t) = v(t)
178        
179        %now the samples for the curvature can be calculated
180        d_d_samples = polyval(d_d, tt);
181        d_dd_samples = polyval(d_dd, tt);
182        s_d_samples = polyval(s_d,tt);
183        
184        kappas_xy= abs(kappa) + abs(d_dd_samples./s_d_samples);
185        kappa_xy_max = max(kappas_xy);
186        
187        if kappa_xy_max > kappaMax
188            %the curvature could be (according to our approximation) to big
189            notD(i) = 1;
190        end
191        
192        
193        vs_x2 = s_d_samples.^2 + d_d_samples.^2;
194        vx_max2 = max(vs_x2);
195        
196        
197        if vx_max2*kappa_xy_max > aOrthMax
198            %the orthogonal acceleration is too much for the grip
199            notD(i) = notD(i)+2; %must be 2. 2+1 = 3 (all), 0 + 2= 2 (only acc.)
200        end
201        
202    end
203    
204    
205end
206
207
208
209%% check for collision
210if polyrootfind
211    
212    for i = 1:n
213        %s(t) should be an increasing function. This is guranteed because we
214        %calculate the min of the 1st derivative above and throw an error if
215        %this is non positive
216        
217        abstandMinusSafety_poly = [0 0 0 dataVeh(2) dataVeh(1)-safetyS]' - PS(:,i);
218        
219        r = roots(abstandMinusSafety_poly);
220        
221        ind = imag(r) == 0;%only real roots
222        r = r(ind);
223        
224        ind = (r>0) & (r < T(i)); %only roots in interval
225        r = r(ind);
226        
227        if (isempty(r)) && (polyval(abstandMinusSafety_poly, T(i)/2) > 0)
228            %all good no collsion possible
229        elseif (isempty(r)) && (polyval(abstandMinusSafety_poly, T(i)/2) <= 0)
230            %fix collsion
231            coll(i) = 1;
232        elseif (length(r) == 1) && (polyval(abstandMinusSafety_poly, r(1)/2) > 0)
233            %one unique possible collision possible
234            d_r = polyval(PD(:,i),r(1));
235            
236            I = dataVeh(3);
237            
238            if (I == -1) && (d_r < safetyD)
239                coll(i) = 1;
240            end
241            
242            if (I ==  1) && (d_r > safetyD)
243                coll(i) = 1;
244            end
245        elseif (length(r) == 1) && (polyval(abstandMinusSafety_poly, r(1)/2) > 0)
246            %there is a collision directly at the start
247            coll(i) = -1;
248        else
249            flag = -3;
250            return;
251        end
252    end
253    
254end
255
256if ~polyrootfind
257    for i = 1:n
258        I = dataVeh(3);%just for readability
259        
260        tt = linspace(0, T(i), nSamples);
261        dd = polyval(PD(:,i),tt);
262        ss = polyval(PS(:,i),tt);
263        
264        for k = 1:nSamples
265            if (I == -1) && (dd(k) < safetyD)
266                abstand = dataVeh(1) + dataVeh(2)*tt(k) - ss(k); %anfangsabstand + geschw. Auto * t - position unser Auto
267                if abstand < safetyS
268                    coll(i) = 1; %more than one collsion doesn't matter
269                end
270            end
271            
272            if (I ==  1) && (dd(k) > safetyD)
273                abstand = dataVeh(1) + dataVeh(2)*tt(k) - ss(k); %anfangsabstand + geschw. Auto * t - position unser Auto
274                if abstand < safetyS
275                    coll(i) = 1; %more than one collsion doesn't matter
276                end
277            end
278        end
279        
280    end
281    
282end
